// export the elements from Quintiq into the sheets classes_I, attributes_I and associations_I
// for the classes, open an editor, and export from left upper pane
//    make sure the classes are sorted by name (otherwise, looking up will be wrong)
// for the attributes, open the from Overview/Attributes
// for the associations
//   run this piece fo code
//   copy the log output in the server output form
//    past it in excel, sheet Classes_I
domain := DomainModel::Domain();
traverse(domain.ModeledRelations(), Elements, association)
  {
  left  := association.LeftSide();
  right := association.RightSide();
  strings := construct(Strings);
  strings.Add(association.Name());
  strings.Add(left.Name());
  strings.Add(left.ObjectDefinition().Name());
  strings.Add(left.Cardinality());
  strings.Add(right.Name());
  strings.Add(right.ObjectDefinition().Name());
  strings.Add(right.Cardinality());
  string := strings.Concatenate(";");
  if ( tolog )
    {
    info('..association', association.Name());  
    info('....left', left.Name(), left.Cardinality(), left.ObjectDefinition().Name());       
    info('....right', right.Name(), right.Cardinality(), right.ObjectDefinition().Name());       
    }
  else
    {
    info(string);
    }
  }
